def train_one_vs_all(X, y, houses):
    """
    Entraîne 4 modèles (un par maison)
    Returns: dict {house_name: theta}
    """
    models = {}
    
    for house in houses:
        # Crée y_binary : 1 si y==house, 0 sinon
        y_binary = (y == house).astype(int)
        
        # Entraîne un modèle
        model = LogisticRegression()
        model.fit(X, y_binary)
        
        # Stocke theta
        models[house] = model.theta
    
    return models
Objectif : Entraîner 4 modèles
Boucle : Pour chaque maison, créer y_binary et fit()
Return : dict {house_name: theta}
☐
Ex 4.3
Fonction predict_multiclass(X, models, houses)
Signature : 
def predict_multiclass(X, models, houses):
    """
    Prédit la classe pour chaque sample
    Returns: array de noms de maisons
    """
    probas = []
    for house in houses:
        theta = models[house]
        proba = sigmoid(X @ theta)
        probas.append(proba)
    
    # Transpose et prend argmax
    probas = np.array(probas).T
    predictions = np.argmax(probas, axis=1)
    
    # Convertit indices en noms
    return np.array([houses[i] for i in predictions])
Méthode : Calculer 4 probas, prendre argmax
Return : Array de noms de maisons

def standardize(X, mean=None, std=None):
    """
    Standardisation : (X - μ) / σ
    Returns: X_scaled, mean, std
    """
    if mean is None:
        mean = np.mean(X, axis=0)
    if std is None:
        std = np.std(X, axis=0)
    
    X_scaled = (X - mean) / std
    return X_scaled, mean, std
Formule : X_scaled = (X - μ) / σ
Si mean=None : Calculer np.mean(X, axis=0)
Return : X_scaled, mean, std
